CREATE TABLE `home_automation`.`light` ( 
`deviceID` INT UNSIGNED NOT NULL, 
`lightID` INT UNSIGNED NOT NULL AUTO_INCREMENT , 
`color` VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , 
`isOpen` TINYINT(1) NOT NULL , 
PRIMARY KEY (`lightID`), 
CONSTRAINT `light_to_device`
FOREIGN KEY (deviceID)
REFERENCES device(deviceID) ON UPDATE CASCADE ON DELETE RESTRICT ) ENGINE = InnoDB;
